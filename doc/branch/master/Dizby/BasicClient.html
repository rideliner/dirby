  # write the other side's remote_uri to the socket
  write(dump_data(@remote_uri))

def send_request(ref, msg_id, *args, &block)
  arr = [ref, msg_id.id2name, args.length, *args, block]
  arr.map! { |ele| dump_data(ele) }
  write(arr.join(''))
end
